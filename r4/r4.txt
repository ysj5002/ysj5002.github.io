	/* 컨테이너 영역의 단 설정 */
	#container { zoom:1; padding-right:200px;}
	
	/* #container에 절대 폭 지정안함 */
	#container:after{content:""; display:block; clear:both;}
	#content { float:left; width:100%; margin-right:-200px;}

	/* 왼쪽 박스에 폭을 100%로 하고, 오른쪽 마진을 -(음수값)으로 원하는 만큼 설정 */
	#aside {float:right; position:relative; width:170px; left:190px;}
	/* #aside가 오른쪽 박스일 경우 float이 right이고, position는 relative로, 오른쪽 밖으로 빼고 싶은 만큼의 양을 left로 위치값을 준다*/
	또는 #aside {position:fixed; width:170px; left:0px; top:0; z-index:99;}
	/* #aside가 왼쪽 박스일 경우는 position을 fixed로 처리*/
